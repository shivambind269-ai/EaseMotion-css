Implementation Quick Start

CSS Usage
style.css & demo.html 
/* 1. Scale-Hover Micro-Interaction Rule */
.em-scale-toast:hover {
  transform: scale(1.05) translateY(-5px);
  box-shadow: 0 30px 60px -12px rgba(0, 0, 0, 0.8), 0 0 35px var(--toast-glow-color);
  z-index: 10;
}

/* 2. Pause Timer Line On Hover */
.em-scale-toast:hover .toast-progress-fill {
  animation-play-state: paused;
}

/* 3. HTML Markup Structure */
<div class="em-toast-container toast-pos-top-right">
  <div class="em-scale-toast toast-variant-success toast-show" role="status" aria-live="polite">
    <div class="toast-body-flex">
      <div class="toast-icon-wrapper">...</div>
      <div class="toast-details">
        <h4 class="toast-title">Artwork Uploaded!</h4>
        <p class="toast-message">High-res render live on Behance.</p>
      </div>
    </div>
  </div>
</div>
